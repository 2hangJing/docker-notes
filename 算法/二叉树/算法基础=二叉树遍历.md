# 二叉树遍历

给定一个二叉树，返回它的 前序、中序、后序、层次遍历。
```javascript
//  二叉树顺序存储数据
let list = [1, 2, 3, null, 5, 6, 7, null, null, 10];
//  二叉树 JS 对象模拟数据
let tree = {
    val: 1,
    left: {
        val: 2,
        left: null,
        right:{
            val: 5,
            left: {
                val: 10,
                left: null,
                right: null
            },
            right: null
        }
    },
    right: {
        val: 3,
        left: {
            val: 6,
            left: null,
            right: null
        },
        right: {
            val: 7,
            left: null,
            right: null
        }
    }
};
//  树结构
//    1
//  /  \
// 2    3
//  \   /\
//   5 6  7
//  /
// 10
```

## 1. 前序遍历
**前序遍历：父节点->左子树->右子树**

* **递归**  
```javascript
let preorderTraversal_recursion = (tree)=>{
    let list = [];
    let recursionFunc = (node)=>{
        if(node === null)return;
        // 注入数据
        list.push(node.val);
        recursionFunc(node.left);
        recursionFunc(node.right);
    }
    recursionFunc(tree)
    return list;
}
preorderTraversal_recursion(tree); // [1, 2, 5, 10, 3, 6, 7]
```

* **迭代**
```javascript
//  迭代
let preorderTraversal_iteration = (tree)=>{
    let list = [];
    // 根元素入栈
    let stack = [tree];
    
    while(stack.length>0){
        // 栈顶元素出栈
        let node = stack.pop();
        if(node === null)continue;
        // 二叉树顺序存储值
        list.push(node.val)
        // 因为前序遍历顺序为：父-> 左-> 右，所以先把右子树入栈
        stack.push(node.right);
        // 再入栈左子树，下次循环时直接从栈顶取出左子树数据
        stack.push(node.left);
    }  
    return list;
}
preorderTraversal_iteration(tree); // [1, 2, 5, 10, 3, 6, 7]
```

## 2. 中序遍历
**左子树->父节点->右子树**
* **递归**  
```javascript
let middleOrderTraversal_recursion = (tree)=>{
    let list = [];
    let recursionFunc = (node)=>{
        if(node === null)return;
        // 递归遍历左子树
        recursionFunc(node.left);
        // 注入数据
        list.push(node.val);
        // 递归遍历右子树
        recursionFunc(node.right);
    }
    recursionFunc(tree)
    return list;
}
middleOrderTraversal_recursion(tree); // [2, 10, 5, 1, 6, 3, 7]
```

* **迭代【自写】**
```javascript
//  迭代
let middleOrderTraversal_iteration = (tree)=>{
    let list = [];
    // 根结点入栈
    let stack = [tree];
    
    while(stack.length>0){
        // 栈顶元素出栈
        let node = stack.pop();
        // 因为前序遍历顺序为：左-> 父-> 右，所以先把右子树入栈
        // 当右子树为 null，此时无需入栈，无意义。
        node.right !== null && stack.push(node.right);
        // 二叉树顺序存储值
        if(node.left !== null){
            // 拷贝一个单独存储数据结点入栈，左子树出栈后此节点出栈释放存储数据
            stack.push({
                val: node.val,
                left: null,
                right: null
            });
            // 再入栈左子树，下次循环时直接从栈顶取出左子树
            stack.push(node.left);
        }else{
            // 顺序存储数据
            list.push(node.val);
        }
    }  
    return list;
}
middleOrderTraversal_iteration(tree); // [2, 10, 5, 1, 6, 3, 7]
```
* **迭代【网上标准算法】**
```javascript
//  迭代
let middleOrderTraversal_iteration = (tree)=>{
    let list= [];
    let stack = [];
    while(stack.length !== 0 || tree != null){

        while(tree !== null){
            stack.push(tree);
            tree = tree.left;
        }

        let node = stack.pop();
        list.push(node.val);
        tree = node.right;
    }
    return list;
}
middleOrderTraversal_iteration(tree); // [2, 10, 5, 1, 6, 3, 7]
```

## 3. 后序遍历
**前序遍历：左子树 -> 右子树 ->父节点**

* **递归**  
```javascript
let preorderTraversal_recursion = (tree)=>{
    let list = [];
    let recursionFunc = (node)=>{
        if(node === null)return;
        
        recursionFunc(node.left);
        recursionFunc(node.right);
        // 注入数据
        list.push(node.val);
    }
    recursionFunc(tree)
    return list;
}
preorderTraversal_recursion(tree); // [10, 5, 2, 6, 7, 3, 1]
```

* **迭代【自写1】**
```javascript
//  迭代
let postorderTraversal_iteration = (tree)=>{
    let list = [];
    // 根结点入栈
    let stack = [tree];
    
    while(stack.length>0){
        // 栈顶元素出栈
        let node = stack.pop();
        if(node.right !== null || node.left !== null){
            // 因为前序遍历顺序为：左-> 右-> 父，所以先把父结点转换成的数据结点入栈
            // 拷贝一个单独存储数据结点入栈，右子树出栈后此节点出栈释放存储数据
            stack.push({
                val: node.val,
                left: null,
                right: null
            });
            node.right !== null && stack.push(node.right);
            node.left !== null && stack.push(node.left);
        }else{
            list.push(node.val);
        }
    }  
    return list;
}
postorderTraversal_iteration(tree); // [10, 5, 2, 6, 7, 3, 1]
```
* **迭代【自写2】**
后序遍历：l -> r -> m，通过左右镜像得到：m -> r -> l。所以只用实现 根 -> 右 -> 左，再通过 unshift 倒叙输出到数组中即实现后续遍历。
```javascript
//  迭代
let postorderTraversal_iteration = (tree)=>{
    let list = [];
    // 根结点入栈
    let stack = [tree];
    while(stack.length>0){
        // 栈顶元素出栈
        let node = stack.pop();
        if(node === null) continue;
        // 通过倒叙输出将 根->右->左 转换成 左->右->根（后续遍历）
        list.unshift(node.val);
        stack.push(node.left);
        stack.push(node.right);
    }  
    return list;
}
postorderTraversal_iteration(tree); // [10, 5, 2, 6, 7, 3, 1]
```

## 4. 层次遍历
#### Ⅰ、广度优先（BFS：Breadth First Search）
* **递归**  
```javascript
let levelOrderTraversal_recursion = (tree)=>{
    let list = [];
    let recursionFunc = queue =>{
        let levelQueue = [];
        list.push(queue.reduce((total, node)=>{
            total.push(node.val);
            node.left && levelQueue.push(node.left);
            node.right && levelQueue.push(node.right);
            return total;
        }, []));
        // 下一层结点队列不为空，递归执行
        levelQueue.length && recursionFunc(levelQueue)
    }
    recursionFunc([tree]);
    return list;
}
levelOrderTraversal_recursion(tree); // [[1], [2,3], [5,6,7], [10]]
```
* **迭代【自写】**
```javascript
//  迭代
let levelOrderTraversal_iteration = (tree)=>{
    let list = [];
    // 根结点入栈
    let queue = [tree];
    while(queue.length !== 0){
        //  当前层结点数据
        let levelQueue = [];
        //  下一层结点队列
        let nextQueue = [];
        queue.forEach(node=>{
            levelQueue.push(node.val);
            node.left && nextQueue.push(node.left);
            node.right && nextQueue.push(node.right);
        });
        list.push(levelQueue);
        //  循环队列赋值为下一层结点队列
        queue = nextQueue;
    }
    return list;
}
levelOrderTraversal_iteration(tree);
```
